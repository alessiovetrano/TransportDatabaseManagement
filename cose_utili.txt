-NEL CASO DI QUERY IN CUI SI CERCA UN VALORE NULLO SI UTILIZZA "IS NULL" E NON " = NULL"
ES. impiegato assente il giorno 4 Aprile
select * from impiegato im join presenza pr on im.cf_impiegato = pr.cf_presenza where Data_presenza = DATE '2022-04-04' AND ora_entrata IS NULL


-NEL CASO DI QUERY IN CUI ABBIAMO BISOGNO DI VEDERE NELLA TABELLA GLI ORARI UTILIZZIAMO IL TO_CHAR con la seguente sintassi:
select cf_impiegato, TO_CHAR(ora_entrata,'HH24:MI') as "Ora_entrata",TO_CHAR(ora_uscita,'HH24:MI') as "Ora_uscita"
from impiegato im join presenza pr on im.cf_impiegato = pr.cf_presenza where Data_presenza = DATE '2022-04-04' 


-QUERY CHE CI DICE QUANTE SPEDIZIONI SONO FATTE PER OGNI VIAGGIO
select cf_viaggio,data_viaggio,p_iva_forn,count(*) from viaggio join spedizione on cf_viaggio = cf_spedizione AND data_viaggio = data_spedizione
group by cf_viaggio,data_viaggio,p_iva_forn;

-QUERY CHE CI DICE QUANTI IMPIEGATI LAVORANO IN UFFICIO
select num_ufficio,count(*) as persone_ufficio from impiegato join ufficio on ufficio_impiegato = num_ufficio
group by ufficio_impiegato


--QUERY ORE LAVORATE IN UN SINGOLO GIORNO
select cf, data_presenza,floor(((ora_uscita - ora_entrata)*24 - 1)) as ore_effettuate from dipendente dip 
join impiegato im on dip.cf = im.cf_impiegato 
join presenza pr on im.cf_impiegato = pr.cf_presenza;
